## 当前状态 vs 目标状态

| 组件 | 当前在哪 | 问题 | 目标 |
|------|---------|------|------|
| **检索器** | `rag/image_search.py`、`rag/text_retrieval.py`、`rag/hybrid_search.py` 裸函数 | 无法统一注册、无法动态切换 | 抽象为 `BaseRetriever`，统一 `retrieve()` 接口 |
| **路由器** | `nodes.py` 的 `plan_node` + `search_node` if-else | 路由逻辑嵌在 Graph 节点里，加新检索器要改代码 | 独立 `Router` 类，支持注册已启用的检索器 |
| **后处理器** | `hybrid_search.py` 内部调 rerank + `nodes.py` 做价格过滤 + `decide_clarify_node` 做门控 | 三步分散在两个文件三个位置，无法统一编排 | 链式 `Pipeline([Rerank, PriceFilter, GenderFilter, ConfidenceGate])` |
| **响应合成器** | `generate_node` 写死 | 不能切换不同回答风格 | `ResponseSynthesizer` 基类 + 可替换策略 |

## 核心代码模板

### ① 检索器基类

```python
from abc import ABC, abstractmethod
from typing import List, Optional
from rag.image_search import SearchResult


class BaseRetriever(ABC):
    """检索器基类：定义统一的检索接口。"""

    @abstractmethod
    async def retrieve(
        self,
        query: str,
        embedding: Optional[List[float]] = None,
        top_k: int = 10,
        filters: Optional[dict] = None,
    ) -> List[SearchResult]:
        ...

    @property
    @abstractmethod
    def name(self) -> str:
        ...


class ImageRetriever(BaseRetriever):
    """图片检索器：CLIP 512d → Qdrant image 向量空间"""

    async def retrieve(self, query, embedding, top_k=10, filters=None):
        from rag.image_search import search_by_image
        loop = asyncio.get_running_loop()
        return await loop.run_in_executor(
            None, lambda: search_by_image(embedding, top_k=top_k)
        )

    @property
    def name(self):
        return "image"


class TextRetriever(BaseRetriever):
    """文本检索器：BGE-M3 1024d → Qdrant text 向量空间，支持 payload 过滤"""

    async def retrieve(self, query, embedding, top_k=10, filters=None):
        from rag.text_retrieval import search_by_text
        loop = asyncio.get_running_loop()
        category = filters.get("category") if filters else None
        return await loop.run_in_executor(
            None, lambda: search_by_text(embedding, top_k=top_k, category_filter=category)
        )

    @property
    def name(self):
        return "text"


class HybridRetriever(BaseRetriever):
    """混合检索器：RRF 融合双路结果 + cross-encoder rerank"""

    async def retrieve(self, query, embedding, top_k=10, filters=None):
        from rag.hybrid_search import hybrid_search
        loop = asyncio.get_running_loop()
        # embedding 应为 {"image": [...], "text": [...]}
        img_emb = embedding.get("image") if isinstance(embedding, dict) else None
        txt_emb = embedding.get("text") if isinstance(embedding, dict) else embedding
        return await loop.run_in_executor(
            None, lambda: hybrid_search(
                image_embedding=img_emb,
                text_embedding=txt_emb,
                query=query,
                top_k=top_k,
            )
        )

    @property
    def name(self):
        return "hybrid"
```

### ② 路由器

```python
from typing import List
from enum import Enum


class InputMode(Enum):
    IMAGE_ONLY = "image_only"
    TEXT_ONLY = "text_only"
    HYBRID = "hybrid"


class QueryRouter:
    """查询路由器：根据输入模态和上下文选择检索器。"""

    def __init__(self):
        self.retrievers: dict[str, BaseRetriever] = {}

    def register(self, retriever: BaseRetriever):
        self.retrievers[retriever.name] = retriever

    def route(self, state: dict) -> tuple[BaseRetriever, dict]:
        """
        返回 (选中的检索器, 检索参数)

        路由策略:
        1. 有图片 + 有文字 → HybridRetriever (RRF)
        2. 仅图片 → ImageRetriever
        3. 仅文字 → TextRetriever
        4. 多轮对话 → 先 query_rewrite，再 TextRetriever
        """
        has_image = bool(state.get("image_id"))
        has_text = bool((state.get("text") or "").strip())
        has_history = bool(state.get("history"))

        filters = {
            "category": state.get("slots", {}).get("category"),
            "tenant_id": state.get("tenant_id", "default"),
        }

        if has_history and has_text:
            # 多轮：改写 query 后用文本检索
            return self.retrievers["text"], {"query": state["rewritten_query"], "filters": filters}

        if has_image and has_text:
            return self.retrievers["hybrid"], {"filters": filters}

        if has_image:
            return self.retrievers["image"], {"filters": filters}

        if has_text:
            return self.retrievers["text"], {"filters": filters}

        return None, {}
```

### ③ 后处理器链

```python
from abc import ABC, abstractmethod


class BasePostprocessor(ABC):
    """后处理器基类"""

    @abstractmethod
    def process(self, candidates: List[dict], state: dict) -> List[dict]:
        ...


class RerankPostprocessor(BasePostprocessor):
    """Cross-encoder 精排"""

    def process(self, candidates, state):
        query = state.get("rewritten_query") or state.get("text", "")
        if not query or len(candidates) <= 1:
            return candidates

        from rag.rerank import rerank
        docs = [_format(c) for c in candidates]
        ranked = rerank(query, docs, top_n=len(candidates))
        return [candidates[r["index"]] for r in ranked]


class PriceFilterPostprocessor(BasePostprocessor):
    """价格范围过滤"""

    def process(self, candidates, state):
        slots = state.get("slots", {})
        budget_min = slots.get("budget_min")
        budget_max = slots.get("budget_max")
        if not budget_min and not budget_max:
            return candidates

        filtered = [c for c in candidates
                    if (budget_min is None or c.get("price", 0) >= budget_min)
                    and (budget_max is None or c.get("price", 0) <= budget_max)]

        return filtered if len(filtered) >= 2 else candidates


class GenderFilterPostprocessor(BasePostprocessor):
    """性别过滤 — ⚠️ 当前缺失，需补充"""

    def process(self, candidates, state):
        slots = state.get("slots", {})
        gender = slots.get("gender")
        if not gender:
            return candidates

        # 数据中 gender 可能在 description 或单独的 gender 字段中
        filtered = [c for c in candidates
                    if gender in (c.get("description", "") + c.get("category", ""))]
        return filtered if filtered else candidates


class ConfidenceGatePostprocessor(BasePostprocessor):
    """多因子置信度门控"""

    def process(self, candidates, state):
        if not candidates:
            state["need_clarify"] = True
            state["clarify_question"] = "未找到匹配商品"
            return candidates

        top_score = candidates[0].get("score", 0)
        second_score = candidates[1].get("score", 0) if len(candidates) >= 2 else 0
        gap = top_score - second_score

        slots = state.get("slots", {})
        has_budget = slots.get("budget_max") is not None
        has_category = slots.get("category") is not None

        if top_score < 0.6:
            state["need_clarify"] = True
            state["clarify_question"] = "匹配度较低，请提供更多条件"
        elif top_score < 0.75 and gap < 0.05 and not (has_budget and has_category):
            state["need_clarify"] = True
            state["clarify_question"] = "有多款接近，请进一步明确需求"

        return candidates


class PostprocessorPipeline:
    """后处理器链：按注册顺序依次执行"""

    def __init__(self):
        self.processors: List[BasePostprocessor] = []

    def add(self, processor: BasePostprocessor):
        self.processors.append(processor)
        return self  # 支持链式调用

    def run(self, candidates, state):
        for processor in self.processors:
            candidates = processor.process(candidates, state)
        return candidates


# 使用示例
pipeline = PostprocessorPipeline() \
    .add(RerankPostprocessor()) \
    .add(PriceFilterPostprocessor()) \
    .add(GenderFilterPostprocessor()) \
    .add(ConfidenceGatePostprocessor())

candidates = pipeline.run(candidates, state)
```

## 面试备忘

> **Q: 你们的 RAG 查询管线是什么架构？**
>
> 四阶段管线：① 路由器根据输入模态（图片/文字/混合）和上下文（多轮对话）选择检索器；② 四种检索器（图片/文本/混合/引用）统一 `retrieve()` 接口，混合检索器内部用 RRF 融合双路结果；③ 后处理器链式执行——先 cross-encoder 精排（gte-rerank-v2），再结构化过滤（价格/性别/品类），最后多因子置信度门控；④ 响应合成器注入候选商品 + 知识引用 + 槽位信息，带约束的 system prompt（显式引用 + 不确定表达 + 推荐解释）流式生成回答。

> **Q: 为什么要把散落的逻辑重构成独立组件？**
>
> 三个原因：① 可测试——每个检索器/后处理器可以独立单测，不用启动整个 LangGraph；② 可扩展——加一个新的检索方式（比如按品牌精确匹配）只需实现 `BaseRetriever` 并注册到路由器；③ 可观测——每个阶段的输入输出独立打点，方便排查"是检索没召回还是 rerank 排错了"。
